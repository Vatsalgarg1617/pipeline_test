pipeline{
     agent any
    
    
   stages{
      stage('Compile Stage'){

     steps{
      withMaven(maven : 'Maven'){
         bat 'cd internal_folder & mvn clean install'
        }
     }
   }
  
    stage('Testing Stage'){

     steps{
      withMaven(maven : 'Maven3.1.1'){
         bat 'mvn test'
        }
     }
   }


     stage('Deployment Stage'){

     steps{
      withMaven(maven : 'Maven3.1.1'){
         bat 'mvn deploy'
        }
     }
   }
      
   }
}
